373template<typename T, typename StorageOrder>
374void Matrix<T, StorageOrder>::compress(Compression const &c)
375{
376 if (compressed)
377 {
378 std::cout << "Matrix is already compressed" << std::endl;
379 return;
380 }
381
382
383 switch (c)
384 {
385 case Compression::CSR:
386 {
387 if (ordering != Order::Row_major)
388 {
389 std::cerr << "only compress to CSR if row-major ordering" << std::endl;
390 return;
391 }
392
393 // keep track of row index change and element
394 std::size_t temp_ind = dynamic_data.cbegin()->first[0];
395 // first element row
396 IA.push_back(temp_ind);
397
398 for (auto it=dynamic_data.cbegin(); it!=dynamic_data.cend(); ++it)
399 {
400 //std::cout << it->first[0] << " " << std::next(it)->first[0];
401 //* value
402 AA.push_back(it->second);
403 //* column
404 JA.push_back(it->first[1]);
405
406 // index of new row
407 if(it->first[0] != temp_ind)
408 {
409 temp_ind = it->first[0];
410
411 // add same element to empty rows
412 auto it_tmp = std::prev(it);
413 for( std::size_t k=it_tmp->first[0]+1; k<temp_ind; ++k )
414 {
415 IA.push_back(AA.size()-1);
416 }
417
418 //* rows
419 temp_ind = it->first[0];
420 IA.push_back(AA.size()-1);
421 }
422 }
423 IA.push_back(AA.size()-1);
424
425 break;
426 }
427
428 case Compression::CSC:
429 {
430 if (ordering != Order::Column_major)
431 {
432 std::cerr << "only compress to CSC if column-major ordering" << std::endl;
433 return;
434 }
435
436 // keep track of row index change and element
437 std::size_t temp_ind = dynamic_data.cbegin()->first[1];
438 // first element row
439 JA.push_back(temp_ind);
440
441 for (auto it=dynamic_data.cbegin(); it!=dynamic_data.cend(); ++it)
442 {
443 //* value
444 AA.push_back(it->second);
445 //* column
446 IA.push_back(it->first[0]);
447
448 // index of new row
449 if(it->first[1] != temp_ind)
450 {
451 temp_ind = it->first[1];
452
453 // add same element to empty rows
454 auto it_tmp = std::prev(it);
455 for( std::size_t k=it_tmp->first[1]+1; k<temp_ind; ++k )
456 {
457 JA.push_back(AA.size()-1);
458 }
459
460 //* rows
461 temp_ind = it->first[1];
462 JA.push_back(AA.size()-1);
463 }
464 }
465 JA.push_back(AA.size()-1);
466
467 break;
468 }
469
470 } // switch(compression)
471
472 // compressed flag
473 compressed = true;
474 compression = c;
475 // clear memory
476 dynamic_data.clear();
477}

484template<typename T, typename StorageOrder>
485void Matrix<T, StorageOrder>::uncompress()
486{
487 if (!compressed)
488 {
489 std::cout << "Matrix is already uncompressed" << std::endl;
490 return;
491 }
492
493 switch (compression)
494 {
495 case Compression::CSR:
496 {
497 if (ordering != Order::Row_major)
498 {
499 std::cerr << "only decompress from CSR if column-major ordering" << std::endl;
500 return;
501 }
502
503 // sizes of arrays
504 std::size_t i_sz = IA.size();
505 std::size_t j_sz = JA.size();
506
507 // insert elements in map
508 for (std::size_t i=0; i<i_sz; ++i)
509 {
510 // use I vector to loop from index i to i+1 in vector J and data
511 for (std::size_t j=IA[i]; j<IA[i+1]; ++j)
512 {
513 // {col, row}, data
514 dynamic_data.insert( { {i, JA[j]}, AA[j]} );
515 }
516 }
517
518 // insert last element
519 dynamic_data.insert( { {nrow-1, JA[j_sz-1]}, AA[j_sz-1]} );
520 }
521
522 case Compression::CSC:
523 {
524 if (ordering != Order::Column_major)
525 {
526 std::cerr << "only decompress from CSC if column-major ordering" << std::endl;
527 return;
528 }
529
530 // sizes of arrays
531 std::size_t i_sz = IA.size();
532 std::size_t j_sz = JA.size();
533
534 // insert elements in map
535 for (std::size_t j=0; j<j_sz; ++j)
536 {
537 // use I vector to loop from index i to i+1 in vector J and data
538 for (std::size_t i=IA[j]; i<IA[j+1]; ++i)
539 {
540 // {col, row}, data
541 dynamic_data.insert( { {j, JA[i]}, AA[i]} );
542 }
543 }
544
545 // insert last element
546 dynamic_data.insert( { {nrow-1, JA[i_sz-1]}, AA[i_sz-1]} );
547
548 break;
549 }
550
551 } //switch(compression)
552
553 AA.clear();
554 JA.clear();
555 IA.clear();
